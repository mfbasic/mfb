NAME
  receive - read a thread message

SYNOPSIS
  thread::receive(t AS Thread OF Msg TO Out, timeoutMs AS Integer = 0) AS Msg
  thread::receive(t AS ThreadWorker OF Msg TO Out, timeoutMs AS Integer = 0) AS Msg

DESCRIPTION
  Reads from the queue selected by the handle type. Thread reads from the
  worker outbound queue. ThreadWorker reads from the worker inbound queue.
  timeoutMs = 0 does not wait for a message. A positive timeout waits up to
  that many milliseconds.

  Unavailable messages fail with ErrNotFound and timeout expiry fails with
  ErrTimeout.

EXAMPLES
  LET message AS String = thread::receive(t)
  LET timed AS String = thread::receive(t, 10)

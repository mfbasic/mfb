NAME
  ln10 - mathematical constant ln(10) as Float

SYNOPSIS
  math::ln10 AS Float

DESCRIPTION
  The natural logarithm of 10 as a Float value, approximately
  2.302585092994046.

EXAMPLES
  LET value AS Float = math::ln10

  LET log10 AS Float = math::log(x) / math::ln10

SEE ALSO
  math::ln10Fixed, math::ln2, math::e, math::log, math::log10

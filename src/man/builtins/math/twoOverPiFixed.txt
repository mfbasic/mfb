NAME
  twoOverPiFixed - mathematical constant 2 / pi as Fixed

SYNOPSIS
  math::twoOverPiFixed AS Fixed

DESCRIPTION
  The mathematical constant 2 / pi rounded to the nearest Fixed value,
  approximately 0.6366197723675814.

EXAMPLES
  LET value AS Fixed = math::twoOverPiFixed

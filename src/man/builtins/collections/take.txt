NAME
    take - return the first items of a list

SYNOPSIS
    collections::take OF T(value AS List OF T, count AS Integer) AS List OF T

DESCRIPTION
    Returns a new list containing the first count items of value. When count is
    greater than or equal to the length of value, the whole list is returned.
    When count is 0 or negative, the empty list is returned. take is total: it
    never fails. value is not modified.

EXAMPLES
    IMPORT io
    IMPORT collections

    FUNC main AS Integer
      LET head AS List OF Integer = collections::take([1, 2, 3, 4], 2)
      io::print(toString(len(head)))
      RETURN 0
    END FUNC

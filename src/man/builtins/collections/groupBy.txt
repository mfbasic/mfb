NAME
    groupBy - group list items into a map by a key function

SYNOPSIS
    collections::groupBy OF T, K, V(value AS List OF T, keyFn AS FUNC(T) AS K, valFn AS FUNC(T) AS V) AS Map OF K TO List OF V

DESCRIPTION
    Builds a map that groups the items of value by keyFn(item). For each group,
    the value is a list of valFn(item) for the items in that group, in their
    original order. The key type K must be comparable. Pass keyFn and valFn as
    named FUNC values.

EXAMPLES
    IMPORT io
    IMPORT collections

    FUNC parityKey(n AS Integer) AS Integer
      RETURN n - (n / 2) * 2
    END FUNC

    FUNC identity(n AS Integer) AS Integer
      RETURN n
    END FUNC

    FUNC main AS Integer
      LET groups AS Map OF Integer TO List OF Integer =
        collections::groupBy([1, 2, 3, 4], parityKey, identity)
      io::print(toString(len(get(groups, 0))))
      RETURN 0
    END FUNC

NAME
    zip - pair items from two lists position-wise

SYNOPSIS
    collections::zip OF A, B(a AS List OF A, b AS List OF B) AS List OF Pair OF A, B

DESCRIPTION
    Returns a new list whose item at each position holds the items at that
    position from a and b, combined into a built-in Pair OF A, B (see
    mfb man types pair). Pairing stops at the shorter of the two inputs, so the
    result length is the smaller of len(a) and len(b). The first field of each
    pair comes from a and the second from b.

EXAMPLES
    IMPORT io
    IMPORT collections

    FUNC main AS Integer
      LET pairs AS List OF Pair OF Integer, String =
        collections::zip([1, 2], ["a", "b"])
      LET p AS Pair OF Integer, String = get(pairs, 0)
      io::print(toString(p.first))
      io::print(p.second)
      RETURN 0
    END FUNC

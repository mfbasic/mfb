NAME
    drop - return a list with the first items removed

SYNOPSIS
    collections::drop OF T(value AS List OF T, count AS Integer) AS List OF T

DESCRIPTION
    Returns a new list containing all but the first count items of value. When
    count is greater than or equal to the length of value, the empty list is
    returned. When count is 0 or negative, the whole list is returned. drop is
    total: it never fails. value is not modified.

EXAMPLES
    IMPORT io
    IMPORT collections

    FUNC main AS Integer
      LET tail AS List OF Integer = collections::drop([1, 2, 3, 4], 2)
      io::print(toString(len(tail)))
      RETURN 0
    END FUNC

NAME
    partition - split a list into matched and unmatched items

SYNOPSIS
    collections::partition OF T(value AS List OF T, predicate AS FUNC(T) AS Boolean) AS Partition OF T

DESCRIPTION
    Splits value in a single pass into the items for which predicate returns
    TRUE and the items for which it returns FALSE. Returns a built-in
    Partition OF T (see mfb man types partition) whose matched field holds the
    items that satisfied predicate and whose unmatched field holds the rest,
    each in original order. Pass predicate as a named FUNC value.

EXAMPLES
    IMPORT io
    IMPORT collections

    FUNC isPos(n AS Integer) AS Boolean
      RETURN n > 0
    END FUNC

    FUNC main AS Integer
      LET result AS Partition OF Integer = collections::partition([-1, 2, -3, 4], isPos)
      io::print(toString(len(result.matched)))
      io::print(toString(len(result.unmatched)))
      RETURN 0
    END FUNC

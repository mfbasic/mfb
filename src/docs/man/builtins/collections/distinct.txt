NAME
    distinct - remove duplicate items, keeping first occurrences

SYNOPSIS
    collections::distinct OF T(value AS List OF T) AS List OF T

DESCRIPTION
    Returns a new list containing the items of value with duplicates removed.
    The first occurrence of each distinct value is kept, in its original
    position; later duplicates are dropped. The element type T must be
    comparable. value is not modified.

EXAMPLES
    IMPORT io
    IMPORT collections

    FUNC main AS Integer
      LET unique AS List OF Integer = collections::distinct([1, 2, 1, 3, 2])
      io::print(toString(len(unique)))
      RETURN 0
    END FUNC
